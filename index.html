Table

Show Tables

SHOW TABLES;

Get Info about Specific Table 

DESCRIBE Users;

Create Table

CREATE Users(
    id INT,
    name VARCHAR(64)
);

Auto Increment Primary/Unique Key 

CREATE TABLEUsers(
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(64)
);

Create Table with Foreign Key 

CREATE TABLE Users(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(64),
    email VARCHAR(64),
    age INT,
    likes-cookies BOOL,
    FOREIGN KEY (student_id)
        REFERENCES student(id)
);

Alter Table

Changing Table Name 

ALTER TABLE Users RENAME Students;

Adding Column to Table 

ALTER TABLE Users
ADD COLUMN shoe_size INT;

Change Column Name 

ALTER TABLE Users
RENAME COLUMN name TO first_name;

Change Column Data Type 

ALTER TABLE Users
MODIFY COLUMN email VARCHAR(128);

Delete Column 

ALTER TABLE Users
DROP COLUMN shoe_size;

Deleting Table 

DROP TABLE Users;
